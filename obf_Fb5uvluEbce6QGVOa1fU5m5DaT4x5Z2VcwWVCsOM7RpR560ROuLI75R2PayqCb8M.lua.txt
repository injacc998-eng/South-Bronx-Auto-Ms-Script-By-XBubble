-- Fungsi untuk menghitung inventory
local function countItems()
    local counts = {
        water = 0,
        sugar = 0,
        gelatin = 0,
        empty = 0,
        smallMarsh = 0,
        mediumMarsh = 0,
        largeMarsh = 0
    }
    
    if player:FindFirstChild("Backpack") then
        for _, item in ipairs(player.Backpack:GetChildren()) do
            if item:IsA("Tool") then
                local name = item.Name:lower()
                if name:find("water") then
                    counts.water = counts.water + 1
                elseif name:find("sugar") then
                    counts.sugar = counts.sugar + 1
                elseif name:find("gelatin") then
                    counts.gelatin = counts.gelatin + 1
                elseif name:find("empty") then
                    counts.empty = counts.empty + 1
                elseif name:find("small.*marshmallow") or (name:find("marshmallow") and name:find("small")) then
                    counts.smallMarsh = counts.smallMarsh + 1
                elseif name:find("medium.*marshmallow") or (name:find("marshmallow") and name:find("medium")) then
                    counts.mediumMarsh = counts.mediumMarsh + 1
                elseif name:find("large.*marshmallow") or (name:find("marshmallow") and name:find("large")) then
                    counts.largeMarsh = counts.largeMarsh + 1
                elseif name:find("marshmallow") and not (name:find("small") or name:find("medium") or name:find("large")) then
                    -- Kalau cuma "Marshmallow Bag" tanpa ukuran, masukin ke small sebagai default
                    counts.smallMarsh = counts.smallMarsh + 1
                end
            end
        end
    end
    
    return counts
end

-- Fungsi update inventory display
local function updateInventoryDisplay()
    local counts = countItems()
    waterText.Text = "Water: " .. counts.water
    sugarText.Text = "Sugar Block: " .. counts.sugar
    gelatinText.Text = "Gelatin: " .. counts.gelatin
    emptyText.Text = "Empty Bag: " .. counts.empty
    smallCount.Text = counts.smallMarsh
    mediumCount.Text = counts.mediumMarsh
    largeCount.Text = counts.largeMarsh
end

-- Auto refresh inventory setiap 1 detik
spawn(function()
    while true do
        wait(1)
        updateInventoryDisplay()
    end
end)

-- Fungsi update status
local function updateStatus(text, timer)
    statusText.Text = "Craft: " .. text
    if timer then
        timerText.Text = "Timer: " .. timer .. "s"
    else
        timerText.Text = "Timer: -"
    end
end

-- Fungsi tekan E
local function pressE()
    game:GetService("VirtualInputManager"):SendKeyEvent(true, Enum.KeyCode.E, false, game)
    wait(0.2)
    game:GetService("VirtualInputManager"):SendKeyEvent(false, Enum.KeyCode.E, false, game)
    wait(0.3)
end

-- Fungsi untuk mencari dan memegang item
local function findAndHoldItem(itemName)
    updateStatus("Mencari " .. itemName)
    
    -- Cari di Backpack
    if player:FindFirstChild("Backpack") then
        for _, item in ipairs(player.Backpack:GetChildren()) do
            if item.Name:lower():find(itemName:lower()) then
                if item:IsA("Tool") then
                    -- Pegang item (equip)
                    player.Character.Humanoid:EquipTool(item)
                    updateStatus("Memegang " .. itemName)
                    wait(0.5)
                    return true
                end
            end
        end
    end
    
    -- Kalau tidak nemu
    updateStatus("⚠️ " .. itemName .. " tidak ada!")
    return false
end

-- Fungsi utama auto craft
local function autoCraft()
    while craftingActive do
        -- STEP 1: Cari dan Pegang Water
        if not findAndHoldItem("water") then
            updateStatus("⚠️ Water habis! Cek inventory")
            wait(5)
            if not craftingActive then break end
        else
            -- Tekan E ke pot
            updateStatus("Masukkan Water ke pot")
            pressE()
            wait(0.5)
            
            -- Tunggu 20 detik
            updateStatus("Memasak Water", 20)
            for i = 20, 1, -1 do
                if not craftingActive then break end
                updateStatus("Memasak Water", i)
                wait(1)
            end
        end
        
        if not craftingActive then break end
        
        -- STEP 2: Cari dan Pegang Sugar Block Bag
        if not findAndHoldItem("sugar") then
            updateStatus("⚠️ Sugar habis! Cek inventory")
            wait(5)
            if not craftingActive then break end
        else
            -- Tekan E ke pot
            updateStatus("Masukkan Sugar ke pot")
            pressE()
            wait(0.5)
        end
        
        if not craftingActive then break end
        
        -- STEP 3: Cari dan Pegang Gelatin
        if not findAndHoldItem("gelatin") then
            updateStatus("⚠️ Gelatin habis! Cek inventory")
            wait(5)
            if not craftingActive then break end
        else
            -- Tekan E ke pot
            updateStatus("Masukkan Gelatin ke pot")
            pressE()
            wait(0.5)
            
            -- Tunggu 45 detik
            updateStatus("Memasak", 45)
            for i = 45, 1, -1 do
                if not craftingActive then break end
                updateStatus("Memasak Sugar+Gelatin", i)
                wait(1)
            end
        end
        
        if not craftingActive then break end
        
        -
